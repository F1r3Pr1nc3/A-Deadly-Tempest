# List of all habitable ringworld sections.
merg_is_hab_ringworld = {
	habitable_structure = yes  # Exlcude some ringworld entities is_artificial = yes
	has_ringworld_output_boost = yes
	NOT = { is_planet_class = pc_habitat }
	OR = {
		is_colonizable = yes
		is_colony = yes
	}
}
